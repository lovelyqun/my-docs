package com.more.fw.m.platform.demo;

import com.more.fw.m.platform.common.bean.ReturnResult;
import org.springframework.stereotype.Service;
import javax.annotation.Resource;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

@Service
public class DemoLoginService {

    // 私钥 demo值仅供demo参考，非实际正式私钥，也不推荐直接copy用于其他正式系统做私钥
    private final static String CONTANTS_SYSTEM_PUB_CRYPT_PRIVATEKEY = "123456";

    @Resource
    DemoLoginDao demoLoginDao;

    /**
     * 登录验证接口
     *
     * @param userName
     * @param passWord
     * @return
     */
    public ReturnResult loginCheck(String userName, String passWord) {
        ReturnResult result = new ReturnResult();
        Map<String, Object> userData = null;

        try {

            /*******************************************************************
             * 此处注释了加密解密传输用户名密码的信息
             * 以及从数据库校验用户密码的功能
             ********************************************************************
             *****************        开    始      *****************************
             *******************************************************************/

            /******************************************************************************
            // 解密报文获取原始TOKEN
            userName = RsaUtil.privateDecrypt(userName, RsaUtil.getPrivateKey(CONTANTS_SYSTEM_PUB_CRYPT_PRIVATEKEY));
            passWord = RsaUtil.privateDecrypt(passWord, RsaUtil.getPrivateKey(CONTANTS_SYSTEM_PUB_CRYPT_PRIVATEKEY));

            // 数据库成不可逆加密
            passWord = MD5Util.encryption(passWord);

             // 根据用户名密码去校验并获取用户信息
             List userList = demoLoginDao.getUsersByUserNameAndPassWord(userName, passWord);
             if (userList != null && userList.size() > 0) {
                userData = (Map)userList.get(0);
             }
            *****************************************************************************/
            /*******************************************************************
             * 此处注释了加密解密传输用户名密码的信息
             * 以及从数据库校验用户密码的功能
             ********************************************************************
             *****************        结    束      *****************************
             *******************************************************************/

            // =========demo代码，模拟输出用户信息 begin=================
            if (userData == null){
                userData =  new HashMap<String, Object>();

                userData.put("userId", "1111");
                userData.put("userName", "admin");
                userData.put("realName", "测试用户");
                userData.put("orgId", "1");
                userData.put("orgName", "测试机构");
                userData.put("loginLanguage", "zhCN");

                // ===========1、可选数据 组织机构列表 begin===============
                List orgList = new ArrayList();

                Map<String, Object> orgMap_1 = new HashMap<String, Object>();
                orgMap_1.put("orgId", "1");
                orgMap_1.put("orgName", "测试集团信息部");
                orgList.add(orgMap_1);

                Map<String, Object> orgMap_2 = new HashMap<String, Object>();
                orgMap_2.put("orgId", "2");
                orgMap_2.put("orgName", "测试集团总经办");
                orgList.add(orgMap_2);

                userData.put("orgList", orgList);
                // ===========1、可选数据 组织机构列表 end===============

                // ===========2、可选数据 角色列表 begin===============
                List roleList = new ArrayList();

                // ===========2-0、可选数据 角色权限全局属性 begin===============
                Map<String, Object> roleMap_1 = new HashMap<String, Object>();
                roleMap_1.put("roleId", "1");
                roleMap_1.put("roleName", "开发员");
                roleMap_1.put("orgId", "1");
                // ===========2-0、可选数据 角色权限全局属性 end===============

                // ===========2-1、可选数据 wisapp系统权限 begin===============
                // 业务权限的结构比较特殊，他是一个对象内存放多个不同的appid为主的key，
                // 每个key对应的是一个业务权限的list
                // 为什么是一个以appid为key的map对象，不是list，是因为这样可以让前端在获取对象时更快，避免遍历判断
                Map<String, Object> busiPermission = new HashMap<String, Object>();
                String appId = "123456";
                List appBusiPermissionList = new ArrayList();
                Map<String, Object>  permissionObject_1 = new HashMap<String, Object>();
                permissionObject_1.put("id","1");
                permissionObject_1.put("code","add");
                appBusiPermissionList.add(permissionObject_1);
                Map<String, Object>  permissionObject_2 = new HashMap<String, Object>();
                permissionObject_2.put("id","2");
                permissionObject_2.put("code","edit");
                appBusiPermissionList.add(permissionObject_2);
                busiPermission.put(appId, appBusiPermissionList);
                roleMap_1.put("busiPermission",busiPermission);
                // ===========2-1、可选数据 wisapp系统权限 end===============

                roleList.add(roleMap_1);
                userData.put("roleList", roleList);
                // ===========2、可选数据 角色列表 end===============

            }
            // =========demo代码，模拟输出用户信息 begin=================
            result.setCode(DemoContants.CONTANTS_SYSTEM_CODE_LOGIN_SUCCESS);
            result.setMsg(DemoContants.CONTANTS_SYSTEM_MSG_LOGIN_SUCCESS);
            result.setData(userData);


        } catch (Exception e) {
            e.printStackTrace();
            result.setCode(DemoContants.CONTANTS_SYSTEM_CODE_LOGIN_ERROR);
            result.setMsg(DemoContants.CONTANTS_SYSTEM_MSG_LOGIN_ERROR);
        }
        return result;
    }

}
